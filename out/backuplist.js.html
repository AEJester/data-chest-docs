const fs = require("fs");

/**
 * Lists all available backups.
 */
const backupList = () => {
    let items = fs.readdirSync("./backups/");
    let finalized = [];
    for (let i = 0; i < items.length; i++) {
        finalized.push(items[i]);
    }
    return finalized;
}

module.exports = backupList;
